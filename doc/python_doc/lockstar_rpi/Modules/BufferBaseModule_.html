 82    async def set_ch_buffer(self, buffer, writer, respond, buffer_one=True):
 83        """helper method to read in buffer from client. Buffer sent to the MC in packets of floor(MCDataPackage.MAX_NBR_BYTES - 100)/4 floats"""
 84        if len(buffer) > (self.buffer_one_size if buffer_one else self.buffer_two_size):
 85            logging.error(f'set_ch_{"one" if buffer_one else "two"}_buffer - buffer too large: {len(buffer)}')
 86            writer.write(BackendResponse.NACK().to_bytes())
 87            await writer.drain()
 88            return False
 89        else:
 90
 91            logging.debug(f'Backend: set ch {"one" if buffer_one else "two"} buffer')
 92            # send buffer in packets of floor(MCDataPackage.MAX_NBR_BYTES - 100)/4 floats
 93            number_of_floats_per_package = floor((MCDataPackage.MAX_NBR_BYTES - 100)/4)
 94            for i in range(0, len(buffer), number_of_floats_per_package):
 95                
 96                mc_data_package = MCDataPackage()
 97                mc_data_package.push_to_buffer('uint32_t', (16 if buffer_one else 17)) # method_identifier
 98                if i == 0:
 99                    mc_data_package.push_to_buffer('bool', False) #overwrite buffer
100                else:
101                    mc_data_package.push_to_buffer('bool', True) #append to buffer
102                    
103                nbr_values_to_read = number_of_floats_per_package
104                if i + number_of_floats_per_package > len(buffer):
105                    nbr_values_to_read = len(buffer) - i
106                
107                mc_data_package.push_to_buffer('uint32_t', nbr_values_to_read)
108                for f in buffer[i:i+nbr_values_to_read]:
109                    mc_data_package.push_to_buffer('float', f)
110                logging.debug(f'send packet of length: {nbr_values_to_read}')
111                await MC.I().write_mc_data_package(mc_data_package)
112                #wait for acknowledgment of reception by MC
113                ack = await self.check_for_ack(writer=None)
114                if not ack:
115                    logging.error(f'set ch {"one" if buffer_one else "two"} buffer: could not send packet!!')
116                    if writer is not None:
117                        writer.write(BackendResponse.NACK().to_bytes())
118                        await writer.drain()
119                        return False
120            
121            if buffer_one:
122                self.buffer_one = buffer.copy()
123            else:
124                self.buffer_two = buffer.copy()
125
126            if writer is not None:
127                writer.write(BackendResponse.ACK().to_bytes())
128                await writer.drain()
129            return True

131    async def initialize_buffers(self, buffer_one_size: int, buffer_two_size: int, chunks_one_size: int, 
132                                chunks_two_size: int, sampling_rate:int, writer, respond=True):
133        """sets the sizes of the buffer_one and buffer two, numbers of chunks for buffer_one 
134        and buffer_two as well as the sampling rate
135
136        
137        :param buffer_one_size (int): number of floats for channel one buffer
138        :param buffer_two_size (int): number of floats for channel two buffer
139        :param chunks_one_size (int): number of chunks in channel one buffer
140        :param chunks_two_size (int): number of chunks in channel two buffer
141        :param sampling_rate (int): sampling rate in Hz
142        :param writer (_type_): _description_
143        :param respond (bool, optional): _description_. Defaults to True.
144        """
145        
146        if buffer_one_size + buffer_two_size > BufferBaseModule_.BUFFER_LIMIT_kBYTES*250:
147            logging.warning(f'initialize_buffers wrong arguments: buffer_sizes: ({buffer_one_size},{buffer_two_size})')
148            if writer is not None:
149                writer.write(BackendResponse.NACK().to_bytes())
150                await writer.drain()
151            return False
152        elif chunks_one_size + chunks_two_size > BufferBaseModule_.MAX_NBR_OF_CHUNKS:
153            logging.warning(f'initialize_buffers wrong arguments: chunksize: ({chunks_one_size},{chunks_two_size})')
154            if writer is not None:
155                writer.write(BackendResponse.NACK().to_bytes())
156                await writer.drain()
157            return False
158        elif chunks_one_size <= 0 or buffer_one_size <= 0 or chunks_two_size <= 0 or buffer_two_size <= 0:
159            logging.warning(f'Both chunk and buffer sizes must be at least 1')
160            if writer is not None:
161                writer.write(BackendResponse.NACK().to_bytes())
162                await writer.drain()
163            return False
164        else:
165            # fraction = Fraction.from_float(sampling_rate / BackendSettings.mc_internal_clock_rate)
166            # fraction = fraction.limit_denominator(BackendSettings.mc_max_counter)
167            # counter_max = fraction.denominator
168            # prescaler = fraction.numerator
169            self.prescaler, self.counter_max = SamplingRate.calculate_prescaler_counter(sampling_rate)
170
171            logging.info(f'initialize: sampling rate: {sampling_rate}, prescaler: {self.prescaler}, counter_max: {self.counter_max}')
172
173            logging.debug('Backend: initialize_buffers')
174            mc_data_package = MCDataPackage()
175            mc_data_package.push_to_buffer('uint32_t', 18) # method_identifier
176            mc_data_package.push_to_buffer('uint32_t', buffer_one_size)
177            mc_data_package.push_to_buffer('uint32_t', buffer_two_size)
178            mc_data_package.push_to_buffer('uint32_t', chunks_one_size)
179            mc_data_package.push_to_buffer('uint32_t', chunks_two_size)
180            mc_data_package.push_to_buffer('uint32_t', self.prescaler)
181            mc_data_package.push_to_buffer('uint32_t', self.counter_max)
182            await MC.I().write_mc_data_package(mc_data_package)
183            sleep(0.1)
184            result = await self.check_for_ack(writer=(writer if respond else None))
185            if result:
186                self.buffer_one_size = buffer_one_size
187                self.buffer_two_size = buffer_two_size
188                self.chunks_one_size = chunks_one_size
189                self.chunks_two_size = chunks_two_size
190                self.sampling_rate = sampling_rate
191            return result

227        """Upload a list of buffer-indices which define the 'chunks': A chunk is a piece of the buffer which the lockstar outputs
228        once it receives a trigger or an output_on(). A chunk is defined by the last buffer-index which still bellongs to this chunk.

233        """

239

242            # send buffer in chunks of MCDataPackage.MAX_NBR_BYTES
243            mc_data_package = MCDataPackage()
244            mc_data_package.push_to_buffer('uint32_t', 20) # method_identifier
245            for f in chunks:
246                mc_data_package.push_to_buffer('uint32_t', f)
247            await MC.I().write_mc_data_package(mc_data_package)
248            result = await self.check_for_ack(writer=(writer if respond else None))
249            if result:
