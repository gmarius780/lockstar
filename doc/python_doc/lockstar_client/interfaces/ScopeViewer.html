41
42        # Setup a timer to trigger the redraw by calling update_plot.
43        self.timer = QtCore.QTimer()
44        self.timer.setInterval(int(1000/refresh_rate))
45        self.timer.timeout.connect(self.update_plot)
46        self.timer.start()
47
48    def update_plot(self):
49        # Drop off the first y element, append a new one.
50        scope_data = asyncio.run(self.client.get_scope_data())
51        
52        if type(scope_data) is dict:
53            self.canvas.axes.cla()  # Clear the canvas.
54            for trace_name in scope_data.keys():
55                trace = scope_data[trace_name]
56                if len(trace) > 0:
57                    self.canvas.axes.plot(self.time_axis, trace, label=trace_name)
58            
59            self.canvas.axes.legend()
60        else:
61            print(scope_data)
62        # Trigger the canvas to update and redraw.
63        self.canvas.draw()
